cordova.define("cordova/plugin/LoginPlugin",
		function(require, exports, module) {
	var exec = require("cordova/exec");
	/**
	 * This Class provides the APIs to authenticate with Unvired Mobile Platform and other backend systems.
	 * @class LoginPlugin 
	 */
	var LoginPlugin = function() {
	};
	// Callbacks
	/**
	 * Called in response to the login Functions.
	 * @callback LoginPlugin~loginQueryCallback
	 */
	/**
	 * Called with the TYPE of login.
	 * @callback LoginPlugin~loginPluginReceiverCallback
	 * @param {Object} result JSON object with the following TYPES : 
	 * <ol><li>SHOW_LOGIN_SCREEN : Show initial login screen/first time login</li>
	 *<li>SHOW_SUBSEQUENT_LOGIN_SCREEN : Show login screen</li>
	 *<li>LOGIN_SUCCESS : Successful login</li>
	 *<li>LOGIN_ERROR : Login failed</li>
	 *<li>AUTH_SUCCESS : First time authentication/login success</li>
	 *<li>AUTH_ERROR : Authentication failed</li>
	 *</ol>
	 */	

	/**
	 * Determines the TYPE of login
	 * @method LoginPlugin.login 	
	 * @param {Object} loginParameters Parameters passed to server to log in
	 * @param {LoginPlugin~loginQueryCallback} loginQueryCallback Success Callback	 

	 */
	LoginPlugin.prototype.login = function(loginParameters,
			loginQueryCallback) {
		var param = [];
		if (loginParameters != null && loginParameters != "")
			param = [ loginParameters ];
		cordova.exec(function(result) {
			loginQueryCallback(result);
		}, function(e) {
			console.log(e);
		}, "LoginPlugin", "login", param);
	};

	/**
	 * Initial authentication and activation.
	 * @method LoginPlugin.authenticateAndActivate 	
	 * @param {Object} customLoginInputString Login parameters passed to the server to for first time activation/authentication. Ex. {"URL" : urlString,"USER_ID" : "MyName","PASSWORD" : "myPwd","DOMAIN" : "myDomain"}
	 * @param {LoginPlugin~loginQueryCallback} loginQueryCallback Success Callback	 
	 * @example LoginPlugin
	            .authenticateAndActivate({"URL" : urlString,"USER_ID" : "MyName","PASSWORD" : "myPwd","DOMAIN" : "myDomain"},
				                         function(result) {
										 });
	 */
	LoginPlugin.prototype.authenticateAndActivate = function(
			customLoginInputString, loginQueryCallback) {
		cordova.exec(function(result) {
			loginQueryCallback(result);
		}, function(e) {
			console.log(e);
		}, "LoginPlugin", "authenticateAndActivate", [ customLoginInputString ]);
	};

	/**
	 * Used when the user settings are already present
	 * @method LoginPlugin.authenticate 	
	 * @param {Object} customLoginInputString Parameters passed to the server to authenticate. Ex. {"USER_ID" : "Myid","PASSWORD" : "myPassword"}
	 * @param {LoginPlugin~loginQueryCallback} loginQueryCallback Success Callback	 
	 * @example LoginPlugin
	            .authenticate({"USER_ID" : "Myid","PASSWORD" : "myPassword"},
				              function(result) {
			                  });
	 */		
	LoginPlugin.prototype.authenticate = function(
			customLoginInputString, loginQueryCallback) {
		cordova.exec(function(result) {
			loginQueryCallback(result);
		}, function(e) {
			console.log(e);
		}, "LoginPlugin", "authenticate", [ customLoginInputString ]);
	};
	/**
	 * Register a function as the callback for the Notification Listener. Note that there can only be one Registered Notification Listener.
	 * @method LoginPlugin.registerNotificationListener
	 * @param {LoginPlugin~loginPluginReceiverCallback} callBack Success Callback
	 */
	LoginPlugin.prototype.registerNotificationListener = function(
			receiverCallback) {

		cordova.exec(function(result) {
			receiverCallback(result);
		}, function(e) {
			console.log(e);
		}, "LoginPlugin", "registerLoginListener", []);
	};

	var loginPlugin = new LoginPlugin();
	module.exports = loginPlugin;
});

	/**
	 * Called with an array of selected data objects.
	 * @callback DatabasePlugin~selectQueryCallback

		}, function(e) {

		}, "DatabasePlugin", "delete", [ query ]);

	// ===========================================UPDATE_QUERY=================================================
	/**
	 * Called with the lid of the updated header Datastructure.
	 * @callback DatabasePlugin~updateQueryCallback 
	 * @param {String} lid Lid of the updated Datastructure.

	 * @method DatabasePlugin.update

	 * @method RuntimePlugin.log 
	 * @param {int} level The Log Level. There are three log levels, Important(7), Error(8) and Debug(9). Logs written with the log level lesser than the default log level of the logger will always get logged. For example: If you set the log level as Debug(9) then logs with Important(7) and Error(8) get logged.
	 * @param {String} sourceClass Class Name
	 * @param {String} sourceMethod Method Name
	 * @param {String} message Log message

	RuntimePlugin.prototype.log = function(level, sourceClass, sourceMethod,

				"level" : level,
				"sourceClass" : sourceClass,
				"sourceMethod" : sourceMethod,
				"message" : message
		};
		cordova.exec(null, function(e) {
			console.log(e);
		}, "RuntimePlugin", "log", [ logMsg ]);
	};

	var runtimePlugin = new RuntimePlugin();
	module.exports = runtimePlugin;

//=====================================================GENERATE_QUERY_STRING_FROM_DATASTRUCTURE==================================================
function getQueryStringFromDS(dataStructure) {

	var query = "";
	var fields = dataStructure.getFIELDS();

	var qString = "";
	var firstelement = true;

	for ( var prop in fields) {

		if (fields[prop] != "") {

			if (!firstelement) {
				qString += " AND ";
			} else {
				firstelement = false;
			}
			qString += prop + "='" + fields[prop] + "'";
		}
	}

	if (qString == "" || qString == null || qString == undefined) {
		query = {
				"tableName" : dataStructure.getTABLE_NAME(),
				"whereClause" : ""
		};
	} else {
		query = {
				"tableName" : dataStructure.getTABLE_NAME(),
				"whereClause" : qString
		};
	}

	return query;
}

//Take INFO MESSAGE ANS BUSINESSENTITY DATA out of response JSONObject
function parseServerResponse(jsonObj) {

	if (jsonObj.hasOwnProperty("IndienceMessage")) {
		var indienceMessage = jsonObj.IndienceMessage;
		if (indienceMessage && indienceMessage.hasOwnProperty("data")) {
			var data = indienceMessage.data;
		}
	}

	if (data && data.hasOwnProperty("Application")) {
		var application = data.Application;
	}
	var msg = "", beData = "";

	// Check for INFO MESSAGES
	if (data && data.hasOwnProperty("InfoMessage")) {
		var infoMsg = data.InfoMessage;

		if (infoMsg instanceof Array) {
			for (var i = 0; i < infoMsg.length; i++) {
				msg += infoMsg[i].message + " ";
			}
		} else {
			msg = infoMsg.message;
		}
	}

	// Check for DATA
	if (application && application.hasOwnProperty("BusinessEntity")) {
		var businessEntity = application.BusinessEntity;
		beData = getBEFromJsonObject(businessEntity);
	}
	return {
		"TYPE" : "S",
		"MESSAGE" : msg,
		"DATA" : beData
	};

}
//Parse BusinessEntities into array of BES
function getBEFromJsonObject(beList) {

	var BE = new Array();

	if (beList && beList != null) {
		// for more than 1 BisinessEntity...
		if (beList instanceof Array) {
			for (var i = 0; i < beList.length; i++) {
				var headerObj = {};
				var iList = new Array();

				// get headers...
				var headerFieldList = beList[i].header.Field;
				var headerName = beList[i].header.name;
				var hfieldObj = {};
				for (var j = 0; j < headerFieldList.length; j++) {
					hfieldObj[headerFieldList[j].name] = headerFieldList[j].content;
				}
				headerObj["HEADER"] = hfieldObj;
				headerObj["NAME"] = headerName;
				hfieldObj = null;

				// get items...
				var itemList = beList[i].item;
				// for more than 1 item...
				if (itemList instanceof Array) {
					for (var k = 0; k < itemList.length; k++) {
						var ifieldObj = {};
						var itemObj = {};
						var itemFieldList = itemList[k].Field;
						var itemName = itemList[k].name;
						for (var n = 0; n < itemFieldList.length; n++) {
							ifieldObj[itemFieldList[n].name] = itemFieldList[n].content;
						}
						itemObj["ITEM"] = ifieldObj;
						itemObj["NAME"] = itemName;
						iList.push(itemObj);
						ifieldObj = null;
						itemObj = null;
					}
				}
				// for only 1 item...
				else if (itemList instanceof Object) {
					var ifieldObj = {};
					var itemObj = {};
					var itemFieldList = itemList.Field;
					var itemName = itemList.name;
					for (var n = 0; n < itemFieldList.length; n++) {
						ifieldObj[itemFieldList[n].name] = itemFieldList[n].content;
					}
					itemObj["ITEM"] = ifieldObj;
					itemObj["NAME"] = itemName;
					iList.push(itemObj);
					ifieldObj = null;
					itemObj = null;
				}
				headerObj["ITEMS"] = iList;
				iList = null;
				BE.push(headerObj);
			}
		}
		// for only 1 BusinessEntity...
		else if (beList instanceof Object) {
			var headerObj = {};
			var iList = new Array();

			// get headers...
			var headerFieldList = beList.header.Field;
			var headerName = beList.header.name;
			var hfieldObj = {};
			for (var j = 0; j < headerFieldList.length; j++) {
				hfieldObj[headerFieldList[j].name] = headerFieldList[j].content;
			}
			headerObj["HEADER"] = hfieldObj;
			headerObj["NAME"] = headerName;
			hfieldObj = null;

			// get items...
			var itemList = beList.item;
			// for more than 1 item...
			if (itemList instanceof Array) {
				for (var k = 0; k < itemList.length; k++) {
					var ifieldObj = {};
					var itemObj = {};
					var itemFieldList = itemList[k].Field;
					var itemName = itemList[k].name;
					for (var n = 0; n < itemFieldList.length; n++) {
						ifieldObj[itemFieldList[n].name] = itemFieldList[n].content;
					}
					itemObj["ITEM"] = ifieldObj;
					itemObj["NAME"] = itemName;
					iList.push(itemObj);
					ifieldObj = null;
					itemObj = null;
				}
			}
			// for 1 item...
			else if (itemList instanceof Object) {
				var ifieldObj = {};
				var itemObj = {};
				var itemFieldList = itemList.Field;
				var itemName = itemList.name;
				for (var n = 0; n < itemFieldList.length; n++) {
					ifieldObj[itemFieldList[n].name] = itemFieldList[n].content;
				}
				itemObj["ITEM"] = ifieldObj;
				itemObj["NAME"] = itemName;
				iList.push(itemObj);
				ifieldObj = null;
				itemObj = null;
			}
			headerObj["ITEMS"] = iList;
			iList = null;
			BE.push(headerObj);
		}
	}
	return BE;
}

function getDataStructure(tableName, result) {

	var dataStructures = [];

	for (var i = 0; i < result.length; i++) {

		var dataStructure = getStructureInstance(tableName);

		var tableFields = result[i].fields;
		for ( var prop in tableFields) {
			if (dataStructure.fields[prop] != undefined) {
				dataStructure.fields[prop] = tableFields[prop];
			}
		}
		dataStructures.push(dataStructure);
	}
	return dataStructures;
};
